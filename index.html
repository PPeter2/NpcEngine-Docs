Quick start

Up in 60 seconds

1
Install via Wally
[dependencies]
NPCEngine = "bs_eryx/npcengine@0.1.0"
2
Create an NPC
local NPCEngine = require(ReplicatedStorage.NPCEngine)

local npc = NPCEngine.new({
    Model = workspace.MyNPC,
    MoveSpeed = 16,
    SleepRange = 120,
})
3
Register states and go
npc:RegisterState("Idle", {
    OnEnter  = function() print("Idle") end,
    OnUpdate = function()
        local target = npc:GetClosestPlayerInRadius(50)
        if target then npc:ChangeState("Chase") end
    end,
    OnExit = function() end,
})

npc:ChangeState("Idle")
